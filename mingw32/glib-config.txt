#!/bin/sh

# glib-config: wrapper around pkg-config for backward compatibility
# with scripts using glib-config
#
# Benoit Laniel <nels@pgroupe.net>

if test "$2" = ""; then
  MODULE="glib-2.0"
  ARGS=$1
else
  case "$1" in
  glib)
    MODULE="glib-2.0"
    ;;
  gmodule)
    MODULE="gmodule-2.0"
    ;;
  gobject)
    MODULE="gobject-2.0"
    ;;
  gthread)
    MODULE="gthread-2.0"
    ;;
  esac
  ARGS=$2
fi

if test "$ARGS" = "--version"; then
  ARGS="--modversion"
fi

pkg-config $MODULE $ARGS
